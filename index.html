from flask import Flask, render_template_string app = Flask(__name__) # Configure external URLs DOWNLOAD_SITE_URL = 'https://www.example.com/your_download_site' TUTORIAL_SITE_URL = 'https://www.example.com/your_tutorial_site' # HTML template with two horizontal buttons centered on screen template = ''' Download Site Download Tutorial ''' @app.route('/') def index(): return render_template_string( template, download_url=DOWNLOAD_SITE_URL, tutorial_url=TUTORIAL_SITE_URL ) if __name__ == '__main__': app.run(host='0.0.0.0', port=5000, debug=True)
